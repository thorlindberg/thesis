\subsection{Declaration}

\begin{Verbatim}[frame=single,numbers=right,framesep=5mm,label=XML Object]
<Object>
    <Property>Content</Property>
</Object>
\end{Verbatim}

\vspace{1mm}

\begin{Verbatim}[frame=single,numbers=right,framesep=5mm,label=JSON Object]
{
    Object : {
        Property : Content
    }
}
\end{Verbatim}

\vspace{1mm}

\begin{Verbatim}[frame=single,numbers=right,framesep=5mm,label=Proto Object]
message Object {
    Optional Type Property = Content;
}
\end{Verbatim}

\subsection{Syntax}

\begin{Verbatim}[frame=single,numbers=right,framesep=5mm,label=XML Object]
<Person>
    <name>1</name>
    <id>2</id>
    <email>3</email>
</Person>
\end{Verbatim}

\vspace{1mm}

\begin{Verbatim}[frame=single,numbers=right,framesep=5mm,label=JSON Object]
{
    "Person": {
        "name":     "1",
        "id":       "2",
        "email":    "3"
    }
}
\end{Verbatim}

\vspace{1mm}

\begin{Verbatim}[frame=single,numbers=right,framesep=5mm,label=Proto Object]
message Person {
    required string name =  1;
    required int32 id =     2;
    optional string email = 3;
}
\end{Verbatim}
